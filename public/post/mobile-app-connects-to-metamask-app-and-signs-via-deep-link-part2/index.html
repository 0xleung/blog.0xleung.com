const onAppWokUp = (event: {url: string}) => {
    setInintUrl(event.url);
};

useEffect(() => {
    Linking.getInitialURL().then(url => {
        if (url) {
        setInintUrl(url);
        }
    });
    Linking.addEventListener('url', onAppWokUp);
    return () => {
        Linking.removeEventListener('url', onAppWokUp);
    };
}, []);
